% SECTION 3: SIGNAL FORMULAS
% =============================================================================

% --- Monotonic Momentum ---

\begin{equation}
h^{(w)}_{i,t} = \frac{1}{w} \sum_{k=0}^{w-1} \mathbf{1}\left(\operatorname{sign}(r_{i,t-k}) = \operatorname{sign}(\mu^{(w)}_{i,t-k})\right)
\end{equation}

\begin{equation}
s^{\mathrm{mon}}_{i,t} = h^{(w)}_{i,t} \cdot \left|\mu^{(w)}_{i,t}\right|
\end{equation}

% --- Active-Return Mean Reversion ---

\begin{equation}
s^{\mathrm{mr}}_{i,t} = -\bar{r}^{(w)}_{i,t}
\end{equation}

% --- Z-Score Reversal ---

\begin{equation}
s^{\mathrm{zrev}}_{i,t} = -\frac{\bar{r}^{(f)}_{i,t} - \mu^{(s)}_{i,t}}{\sigma^{(s)}_{i,t}}
\end{equation}

% --- Residual Mean Reversion ---

\begin{equation}
s^{\mathrm{res}}_{i,t} = -\bar{\tilde{r}}^{(w)}_{i,t}
\end{equation}

% --- Distance Pairs Mean Reversion ---

\begin{equation}
z_{i,t} = \frac{d_{i,t} - \bar{d}^{(w)}_{i,t}}{\sigma^{(w)}(d_{i,t})} \qquad s^{\text{pairs}}_{i,t} = -\operatorname{clip}(z_{i,t},\, -2,\, 2)
\end{equation}

% --- Sector-Relative Mean Reversion ---

\begin{equation}
s^{\text{sector}}_{i,t} = -\overline{(r_{i,t} - r^{\text{sector}}_{g(i),t})}^{(w)}
\end{equation}

% --- Gap Reversion ---

\begin{equation}
s^{\text{gap}}_{i,t} = -\tilde{r}_{i,t-1}
\end{equation}
